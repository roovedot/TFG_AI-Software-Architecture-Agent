
## Entorno DEV (Ollama local + GPU)

# Levantar
sudo docker compose --env-file .env.dev --profile dev up -d

# Bajar
sudo docker compose --env-file .env.dev --profile dev down

# Logs
sudo docker compose --env-file .env.dev --profile dev logs -f
sudo docker compose --env-file .env.dev --profile dev logs -f api
sudo docker compose --env-file .env.dev --profile dev logs -f ollama


## Entorno PROD (OpenAI / Anthropic)

# Levantar
sudo docker compose --env-file .env.prod up -d

# Bajar
sudo docker compose --env-file .env.prod down

# Logs
sudo docker compose --env-file .env.prod logs -f


## Ollama (solo dev)

# Ver modelos descargados
sudo docker compose --env-file .env.dev --profile dev exec ollama ollama list

# Descargar modelo desde HuggingFace (CDN alternativo si Cloudflare está bloqueado)
sudo docker compose --env-file .env.dev --profile dev exec ollama ollama pull hf.co/bartowski/Llama-3.2-3B-Instruct-GGUF:Q4_K_M

# Crear alias de modelo
sudo docker compose --env-file .env.dev --profile dev exec ollama ollama cp "hf.co/bartowski/Llama-3.2-3B-Instruct-GGUF:Q4_K_M" llama3.2:3b

# Probar modelo interactivo
sudo docker compose --env-file .env.dev --profile dev exec ollama ollama run llama3.2:3b "hola"
