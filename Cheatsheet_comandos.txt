
## Entorno LOCAL (Ollama local + GPU)

# Levantar
sudo docker compose --profile local up -d

# Bajar
sudo docker compose --profile local down

# Logs
sudo docker compose --profile local logs -f
sudo docker compose --profile local logs -f api
sudo docker compose --profile local logs -f ollama


## Entorno PROD (OpenAI / Anthropic, sin Ollama)

# Levantar (cambiar ENVIRONMENT=production en .env)
sudo docker compose up -d

# Bajar
sudo docker compose down

# Logs
sudo docker compose logs -f


## Ollama (solo entorno local)

# Ver modelos descargados
sudo docker compose --profile local exec ollama ollama list

# Descargar modelo desde HuggingFace (CDN alternativo si Cloudflare esta bloqueado)
sudo docker compose --profile local exec ollama ollama pull hf.co/bartowski/Llama-3.2-3B-Instruct-GGUF:Q4_K_M

# Crear alias de modelo
sudo docker compose --profile local exec ollama ollama cp "hf.co/bartowski/Llama-3.2-3B-Instruct-GGUF:Q4_K_M" llama3.2:3b

# Probar modelo interactivo
sudo docker compose --profile local exec ollama ollama run llama3.2:3b "hola"
